kaborArimaAuto<-function(myvector,countDaysInYear,year,dayOfYear,forecastDuration){
	myts <-ts(myvector,  freq=countDaysInYear, start=c(year,dayOfYear ))
	mytsarima <- auto.arima(myts)
	mytimeforecast1 <-forecast(mytsarima, h=forecastDuration)
	myobj <- as.numeric(mytimeforecast1$mean)
	return (myobj)
}
